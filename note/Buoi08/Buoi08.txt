1. visited, hover, active

// Màu hiển thị của đường link
/* visited link */
a:visited {
  color: purple;
}

// Màu khi đưa chuột vào link
/* mouse over link */
a:hover {
  color: red;
}

// Màu khi nhấp vào link
/* selected link */
a:active {
  color: yellow;
}
